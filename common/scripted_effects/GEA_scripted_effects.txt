Burma_ger_click_effect = {
	if = {
		limit = {
			NOT = {
				has_country_flag = Burma_ger_OPEN
			}
		}
		set_country_flag = Burma_ger_OPEN
	}
	
	else_if = {
		limit = {
			has_country_flag = Burma_ger_OPEN
		}
		clr_country_flag = Burma_ger_OPEN
	}
}
