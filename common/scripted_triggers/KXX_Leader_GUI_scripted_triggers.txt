KXX_uses_leader_gui = {
	OR = {
		AND = {
			tag = GRU
			has_country_flag = GRU_GOEBB_GORING_VISIBLE
			NOT = {
				has_country_flag = GRU_pm_is_hos
			}
		}

		GRU_has_leader_GUI = yes
		has_country_flag = KXX_LEADER_GUI_VISIBLE
	}
	NOT = {
		has_country_flag = KXX_LEADER_GUI_NOT_VISIBLE
	}
}
